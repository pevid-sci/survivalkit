1. Python Base: Tipos, Lógica e Controle

Tipos & Conversões
type(x), id(x) # Retorna classe e endereço de memória int(x), float(x) # Converte para inteiro / decimal str(x), bool(x) # Converte para texto / booleano isinstance(x, int)# True se x for inteiro round(3.1415, 2) # Arredonda para 2 casas (3.14)
Operadores Lógicos & Identidade
a == b, a != b # Igual, Diferente a > b, a <= b # Maior, Menor ou igual a and b, a or b # E, Ou lógico not a # Inverte booleano a is b, a is not b# Compara se é o mesmo objeto na memória x in lista # True se x estiver contido no iterável
Controle de Fluxo (If / Else)
if p_valor < 0.05: print("Sig") elif p_valor < 0.10: print("Tend") else: print("NS") # Operador Ternário status = "Alta" if vivo else "Óbito"
Loops (For / While)
for item in sequencia: if item == "pula": continue if item == "para": break while contador < 10: contador += 1 for i, val in enumerate(lista): # Loop com índice numérico for a, b in zip(list1, list2): # Itera em 2 listas simultaneamente
Tratamento de Erros (Try/Except)
try: resultado = 10 / 0 except ZeroDivisionError: print("Erro matematico") except Exception as e: print(f"Erro geral: {e}") finally: print("Roda sempre, dando erro ou não")

2. Estruturas de Dados e Comprehensions

Strings (Textos)
s[0:5], s[::-1] # Fatiamento e inversão s.upper(), s.lower()# Maiúsculo, Minúsculo s.strip(), s.split()# Remove espaços laterais, Quebra em lista s.replace(a, b) # Substitui 'a' por 'b' s.startswith("A") # True se começa com A "-".join(lista) # Junta lista de strings com hífens f"N={tamanho}" # F-String (Interpolação de variáveis)
Listas [ ]
l.append(x) # Adiciona no final l.extend([x, y]) # Adiciona múltiplos itens no final l.insert(0, x) # Insere na posição 0 l.pop(), l.pop(0) # Remove o último / Remove o primeiro l.remove(x) # Remove a 1ª ocorrência de x l.sort() # Ordena in-loco (crescente) l.count(x) # Conta quantas vezes x aparece
Dicionários { }
d[chave] = valor # Adiciona ou atualiza d.get(chave, 0) # Busca. Se não achar, retorna 0 (evita erro) d.keys() # Lista as chaves d.values() # Lista os valores d.items() # Tuplas de (chave, valor) d.update(d2) # Funde com outro dicionário d2 d.pop(chave) # Remove e retorna o valor
Tuplas () e Sets {}
t = (1, 2) # Imutável. Usada p/ dados fixos/coordenadas s = {1, 2, 2} # Conjunto. Vira {1, 2} (remove duplicatas) s.add(3) # Adiciona no set s1 | s2 # União de dois sets s1 & s2 # Intersecção (o que tem em ambos) s1 - s2 # Diferença (tem no s1 mas não no s2)
List & Dict Comprehensions
# [expressão for item in iterável if condição] quadrados = [x**2 for x in range(10)] pares = [x for x in lista if x % 2 == 0] # {chave: valor for item in iterável} dic_quad = {x: x**2 for x in range(5)} # If/Else dentro da list comprehension rotulo = ["Pos" if x>0 else "Neg" for x in lista]

Inspeção do DataFrame
df.head(), df.tail() # Primeiras/Últimas linhas df.sample(5) # 5 linhas aleatórias df.info() # Estrutura, Dtypes, Memória df.describe(include="all")# Estatística descritiva df.shape # Tupla (linhas, colunas) df.columns # Index com nomes das colunas df.index # Index das linhas df.dtypes # Tipos de dados de cada coluna

5. PANDAS: Filtros, Seleção (Slicing) e Consultas

Seleção Básica (Colunas e Linhas)
df['idade'] # Retorna Series (1 col) df[['idade', 'sexo']] # Retorna DataFrame (2 cols) df.iloc[0] # Primeira linha inteira df.iloc[-1] # Última linha inteira df.iloc[0:5, 0:3] # Linhas 0 a 4, Colunas 0 a 2 df.loc[100, 'idade'] # Pega linha cujo rótulo/index é 100, coluna idade
Filtros Booleanos (Condicionais)
df[df['idade'] > 60] # Simples df[(df['A']>1) & (df['B']<5)] # Múltiplas condições (E) df[(df['A']>1) | (df['B']<5)] # Múltiplas condições (OU) df[~df['obito'] == 1] # Negativo (Não é óbito) df[df['UF'].isin(['RS', 'SP'])] # Está contido na lista df[df['UF'].isnull()] # Apenas linhas com UF vazia
Query (Filtro por String) e Nlargest
df.query('idade > 60 and UF == "RS"') # Mais limpo de ler limite = 50 df.query('idade > @limite') # Usa variável externa com @ df.nlargest(5, 'salario') # 5 linhas com maiores salários df.nsmallest(5, 'idade') # 5 linhas com menores idades

6. PANDAS: Limpeza, Nulos e Modificação de Tabelas

Tratamento de Missing Data (NaN)
df.isna().sum() # Soma de nulos por coluna df.dropna() # Remove qualquer linha c/ NaN df.dropna(subset=['obito'])# Remove se NaN apenas na col 'obito' df.fillna(0) # Troca todos NaN por 0 df['PAS'].fillna(df['PAS'].mean()) # Imputa c/ média df.ffill(), df.bfill() # Preenche com valor anterior / próximo df.interpolate() # Interpolação matemática p/ séries
Modificando Colunas & Tipos
df.rename(columns={'old': 'new'}) # Muda nome df.drop(columns=['A', 'B']) # Deleta colunas df.drop(index=[0, 1]) # Deleta linhas 0 e 1 df['A'] = df['A'].astype(str) # Converte tipo df['data'] = pd.to_datetime(df['data']) # Converte p/ Data df['num'] = pd.to_numeric(df['num'], errors='coerce') # Força numero, vira NaN se errar
Duplicatas & Ordenação
df.duplicated().sum() # Conta linhas duplicadas df.drop_duplicates() # Remove linhas 100% iguais df.drop_duplicates(subset=['id'], keep='last') # Mantém último registro df.sort_values('idade', ascending=False) # Ordem decrescente df.sort_values(['grupo', 'idade'])# Ordena por múltiplas df.reset_index(drop=True) # Refaz o índice numérico 0,1,2...
Matemática Vetorizada e Lógica Linha-a-Linha
df['imc'] = df['peso'] / (df['alt']**2) # Cria coluna matematicamente df['grupo'].replace({'A':1, 'B':0}) # Dicionário de substituição import numpy as np df['risco'] = np.where(df['idade']>60, 'Alto', 'Baixo') # Equivalente ao ifelse() do R df['calc'] = df.apply(lambda row: func(row['A'], row['B']), axis=1) # Aplica func complexa nas linhas

7. PANDAS: Métodos `.str` e `.dt` (Acessadores)

O Acessador .str (Textos)
df['nome'] = df['nome'].str.upper() # Tudo maiúsculo df['nome'].str.strip() # Limpa espaços em branco df['nome'].str.contains('Silva') # Filtro: Contém a palavra? df['nome'].str.startswith('A') # Inicia com A? df['cod'].str.extract(r'(\d+)') # Extrai apenas números via Regex df['sobrenome'] = df['nome'].str.split(' ').str[-1] # Pega a última palavra
O Acessador .dt (Séries Temporais)
df['ano'] = df['data'].dt.year # Extrai o ano inteiro df['mes'] = df['data'].dt.month # Extrai o mês (1-12) df['dia_sem'] = df['data'].dt.day_name() # Nome do dia (Monday) df['duracao_dias'] = (df['data_alta'] - df['data_int']).dt.days # Calc tempo df.set_index('data').resample('M').sum() # Agrupa por Mês (Mensal)

8. PANDAS: Agrupamento, Cruzamento e Reshape

Groupby e Estatísticas (Summarise)
df['cat'].value_counts(normalize=True)# Frequência relativa (%) df.groupby('grupo')['idade'].mean() # Média de 1 coluna por grupo df.groupby(['grupo', 'sexo']).size() # Contagem em grupos duplos # Agg múltiplo (A "Tabela 1" do artigo) df.groupby('braco').agg({ 'idade': ['mean', 'std'], 'pas': 'median', 'obito': 'sum' }).reset_index()
Merge, Join e Concat
# Empilhar tabelas verticalmente (rbind) df_full = pd.concat([df1, df2], ignore_index=True) # Cruzamento de tabelas (merge / join) # how: 'left', 'right', 'inner', 'outer' pd.merge(pacientes, exames, on="id", how="left") pd.merge(df1, df2, left_on="id_pac", right_on="pront")
Reshape: Pivot e Melt (Wide vs Long)
# Transforma formato LONGO para LARGO (Tabela Dinâmica) df_wide = df.pivot_table(index='id', columns='tempo', values='pas') # Transforma formato LARGO para LONGO (Unpivot) df_long = pd.melt(df_wide, id_vars=['id', 'idade'], value_vars=['pas_t0', 'pas_t1'], var_name='tempo', value_name='pressao')
Bins (Categorização de Numéricas)
# Cria faixas etárias fixas df['faixa'] = pd.cut(df['idade'], bins=[0, 18, 60, 100], labels=['Jovem', 'Adulto', 'Idoso']) # Cria quartis perfeitos (Mesma quantidade em cada grupo) df['quartil_renda'] = pd.qcut(df['renda'], q=4)

9. Bioestatística, Epi e Modelagem (Scipy, Statsmodels, Lifelines)

Scipy.Stats (Testes Clássicos)
from scipy import stats stats.ttest_ind(grupoA, grupoB) # Teste T Independente stats.mannwhitneyu(grupoA, grupoB) # U de Mann-Whitney (Não param) stats.chi2_contingency(tabela_cruz) # Qui-Quadrado stats.fisher_exact(tabela_2x2) # Exato de Fisher stats.pearsonr(x, y) # Correlação de Pearson
Statsmodels (Regressões Estilo R)
import statsmodels.formula.api as smf # Regressão Linear Simples/Múltipla mod = smf.ols("pas ~ idade + C(sexo)", data=df).fit() print(mod.summary()) # Regressão Logística (Odds Ratio) mod_log = smf.logit("obito ~ tratamento + idade", data=df).fit() odds_ratios = np.exp(mod_log.params) # Extrai os ORs
Lifelines (Análise de Sobrevivência)
from lifelines import KaplanMeierFitter, CoxPHFitter # Kaplan-Meier (Curva) kmf = KaplanMeierFitter() kmf.fit(duracoes=df['tempo'], event_observed=df['evento']) kmf.plot_survival_function() # Modelo de Cox Proporcional (Hazard Ratios) cph = CoxPHFitter() cph.fit(df, duration_col='tempo', event_col='evento') cph.print_summary() cph.plot() # Plota o Forest Plot dos HRs
