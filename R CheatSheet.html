R Survival Kit

🆘 Operadores e Ajuda (R Base)

? ou help() (base): Abre a página oficial de documentação de uma função específica
?mean # abre o painel de ajuda explicando como usar a função mean
?? ou help.search() (base): Pesquisa por um termo em toda a documentação de todos os pacotes instalados
??concordance # busca qualquer função ou pacote que mencione "concordance"
== e != (base): Operadores lógicos de "Igual a" e "Diferente de" (is not)
filter(df, sexo == "F" & status != "Inativo") # filtra onde sexo é F e status não é Inativo
>= e <= (base): Operadores lógicos de "Maior/Menor ou igual a"
filter(df, idade >= 18) # filtra observações onde a idade é maior ou igual a 18
%% (base): Operador de módulo (retorna o resto de uma divisão)
10 %% 3 # retorna 1, pois 10 dividido por 3 tem resto 1
%in% (base): Operador de pertencimento (verifica se um valor está dentro de um vetor)
filter(df, estado %in% c("RS", "SC", "PR")) # mantém linhas onde o estado é um desses três
|> ou %>% (base / magrittr): Operador Pipe (passa o resultado da esquerda para a função da direita)
df |> filter(idade > 18) |> summarize(media = mean(salario)) # encadeia operações de forma fluida

🗂️ Manipulação de Dados, Importação e Exportação (Tidyverse, haven, rvest)

tibble (tibble): Cria um data frame moderno, otimizado e fácil de ler
tibble(x = 1:5, y = x * 2) # cria uma tabela com as colunas x e y
read_csv (readr): Lê um arquivo de texto separado por vírgulas para um data frame
read_csv("dados.csv") # importa o arquivo dados.csv para o R
write_csv (readr): Salva um data frame em um arquivo CSV no computador
write_csv(df, "resultado.csv") # exporta o data frame df para resultado.csv
read_dta (haven): Importa conjuntos de dados de softwares como SPSS, Stata e SAS
read_dta("banco.dta") # lê arquivo nativo do Stata mantendo os rótulos originais
read_html (rvest): Realiza web scraping para extrair dados estruturados da internet
read_html(url) |> html_elements(".classe") # raspa os elementos HTML de uma página
filter (dplyr): Filtra linhas de um data frame com base em condições lógicas
filter(df, idade > 18) # mantém apenas as linhas onde a idade é maior que 18
arrange (dplyr): Ordena as linhas de um data frame com base em uma ou mais colunas
arrange(df, desc(salario)) # ordena o data frame pelo salário em ordem decrescente
select (dplyr): Seleciona e filtra colunas (variáveis) específicas de um data frame
select(df, nome, idade) # mantém apenas as colunas nome e idade no resultado
mutate (dplyr): Cria novas colunas ou modifica colunas existentes
mutate(df, dobro = valor * 2) # cria uma nova coluna 'dobro' com o valor x2
summarize (dplyr): Resume os dados reduzindo-os a um único valor (ex: média, soma)
summarize(df, media = mean(valor)) # calcula e retorna a média da coluna valor
group_by (dplyr): Agrupa os dados para que operações sejam feitas por grupo
group_by(df, categoria) # agrupa o data frame pela variável categoria
count (dplyr): Conta rapidamente o número de observações para cada grupo
count(df, sexo) # retorna a quantidade total de linhas para cada sexo
left_join (dplyr): Junta dois data frames mantendo todas as linhas do primeiro
left_join(df1, df2, by = "id") # adiciona colunas do df2 ao df1 cruzando pela chave id
pivot_longer (tidyr): Transforma múltiplas colunas em linhas (formato longo)
pivot_longer(df, cols = a:c, names_to = "k", values_to = "v") # comprime as colunas em linhas
pivot_wider (tidyr): Transforma linhas em múltiplas colunas (formato largo)
pivot_wider(df, names_from = k, values_from = v) # espalha os valores em novas colunas
separate (tidyr): Separa o texto de uma única coluna em várias colunas
separate(df, data, into = c("ano", "mes", "dia"), sep = "-") # divide a coluna data em três
unite (tidyr): Combina o conteúdo de várias colunas em uma única coluna
unite(df, data_full, ano, mes, dia, sep = "-") # une as três colunas com traço
drop_na (tidyr): Remove todas as linhas que contêm valores ausentes (NA)
drop_na(df) # exclui qualquer linha do data frame que possua NA
map (purrr): Aplica uma função a cada elemento de uma lista ou vetor
map(lista, mean) # retorna uma lista com a média calculada para cada elemento
str_detect (stringr): Verifica se um padrão de texto existe dentro de uma string
str_detect(texto, "R") # retorna TRUE para os elementos que contêm "R"
str_replace (stringr): Substitui a primeira ocorrência de um texto por outro
str_replace(texto, "ruim", "bom") # substitui "ruim" por "bom" na string
ymd (lubridate): Converte texto em data reconhecida pelo R no formato Ano-Mês-Dia
ymd("2023-12-31") # transforma string na data oficial 2023-12-31
fct_reorder (forcats): Reordena níveis de um fator com base em outra variável
fct_reorder(fator, valores, mean) # ordena o fator pela média dos valores

📊 Visualização de Dados e Relatórios (ggplot2, ggpubr, patchwork, gtsummary, eclipseplot)

ggplot (ggplot2): Inicializa um objeto gráfico e define os dados utilizados
ggplot(data = df) # cria a base vazia do gráfico apontando para o data frame
aes (ggplot2): Define a estética, mapeando as variáveis aos eixos e cores
aes(x = idade, y = salario) # mapeia idade no eixo x e salario no eixo y
geom_point (ggplot2): Adiciona uma camada geométrica de pontos (dispersão)
geom_point() # plota um gráfico de dispersão com os dados definidos
geom_line (ggplot2): Adiciona uma camada geométrica de linhas conectadas
geom_line() # conecta pontos sequencialmente (útil para séries temporais)
geom_histogram (ggplot2): Cria um histograma para visualizar distribuições contínuas
geom_histogram(binwidth = 5) # cria histograma com blocos de tamanho 5
geom_boxplot (ggplot2): Cria um gráfico de caixa para exibir distribuições e outliers
geom_boxplot() # exibe a mediana, quartis e outliers da variável
facet_wrap (ggplot2): Divide o gráfico em múltiplos painéis por categoria
facet_wrap(~ categoria) # cria um mini-gráfico separado para cada categoria
ggboxplot (ggpubr): Cria boxplots focados em publicações científicas
ggboxplot(df, x = "grupo", y = "valor", add = "jitter") # boxplot com pontos sobrepostos
patchwork (patchwork): Combina múltiplos gráficos do ggplot2 em uma única figura
grafico1 + grafico2 # alinha os dois gráficos lado a lado
tbl_summary (gtsummary): Cria tabelas prontas para publicação (Tabela 1)
tbl_summary(dados, by = grupo) # gera tabela descritiva elegante por grupo
eclipse_plot (eclipseplot): Cria visualizações em formato de eclipse para ensaios
eclipse_plot(dados) # gera um gráfico de eclipse para interpretação de resultados

🏗️ Estruturas e R Base

c (base): Combina valores simples ou elementos em um único vetor
c(1, 2, 3, 4) # cria um vetor numérico com 4 elementos
data.frame (base): Cria uma estrutura tabular clássica para armazenar dados mistos
data.frame(id = 1:3, nome = c("A", "B", "C")) # cria data frame tradicional
matrix (base): Cria uma matriz matemática bidimensional (mesmo tipo de dado)
matrix(1:9, nrow = 3, ncol = 3) # cria uma matriz 3x3 preenchida de 1 a 9
list (base): Cria uma lista que pode conter diferentes tipos e tamanhos de objetos
list(numero = 1, texto = "A", vetor = c(1, 2)) # agrega tipos diferentes
factor (base): Converte um vetor em dados categóricos
factor(c("baixo", "alto", "baixo")) # define níveis para as categorias
summary (base): Retorna um resumo estatístico rápido do objeto
summary(df) # mostra resumo (média, quartis) para as colunas
str (base): Exibe a estrutura interna de um objeto (tipos de colunas)
str(df) # mostra o tipo de cada coluna e amostra inicial
head (base): Retorna as primeiras linhas de um objeto
head(df, n = 5) # exibe apenas as 5 primeiras linhas
table (base): Cria uma tabela de contingência (contagem de frequências)
table(df$categoria) # conta quantas vezes cada categoria aparece
tapply (base): Aplica uma função estatística a subconjuntos de um vetor
tapply(df$salario, df$cargo, mean) # média salarial por cargo
apply (base): Aplica função nas linhas ou colunas de matriz/data frame
apply(matriz, 1, sum) # calcula a soma de cada linha (margem 1)
lapply (base): Aplica função sobre lista/vetor e retorna sempre uma lista
lapply(lista, length) # retorna o comprimento de cada item na lista
plot (graphics): Função genérica nativa para criar gráficos
plot(x, y) # cria um gráfico simples de dispersão x vs y
seq (base): Gera uma sequência regular de números
seq(from = 1, to = 10, by = 2) # gera: 1, 3, 5, 7, 9
rep (base): Replica elementos de vetores ou listas
rep(x = 1, times = 5) # cria vetor com o número 1 repetido 5 vezes
paste (base): Concatena vetores de texto com um separador
paste("Olá", "Mundo", sep = " ") # retorna "Olá Mundo"
ifelse (base): Avalia condição lógica de forma vetorizada
ifelse(x > 0, "Positivo", "Negativo") # retorna valor baseado em x
subset (base): Filtra elementos de vetores ou data frames
subset(df, idade > 30) # retorna linhas onde idade > 30
is.na (base): Verifica quais valores são ausentes (NA)
is.na(x) # retorna TRUE/FALSE para a posição de cada NA
na.omit (base): Remove linhas com valores ausentes do conjunto
na.omit(df) # limpa o data frame de qualquer linha com NA

🔬 Modelagem, Estatística Médica, Meta-Análise e Inferência Causal

lm (stats): Ajusta um modelo de regressão linear simples ou múltipla
lm(y ~ x1 + x2, data = df) # ajusta y em função de x1 e x2
glm (stats): Ajusta modelos lineares generalizados (Logística, Poisson)
glm(y ~ x, family = binomial, data = df) # regressão logística
t.test (stats): Realiza o teste estatístico T de Student para médias
t.test(x, y) # compara médias de x e y para diferença significativa
tidy (broom): Limpa a saída de modelos em tabelas arrumadas (tibbles)
tidy(modelo_logistico) # extrai coeficientes e p-valores em formato tabular
lmer (lme4): Ajusta modelos de efeitos mistos (hierárquicos/longitudinais)
lmer(resultado ~ tratamento + (1 | paciente), data = d) # intercepto aleatório
cac.raw4 (irrCAC): Calcula coeficientes de concordância (Gwet AC1)
cac.raw4(dados) # estatísticas de concordância para múltiplos avaliadores
rma (metafor): Ferramentas avançadas para meta-análises e meta-regressões
rma(yi, vi, data = dados) # modelo de meta-análise de efeitos aleatórios
metabin (meta): Realiza meta-análises para desfechos binários (OR, RR)
metabin(e.ev, e.n, c.ev, c.n, data = d) # pooling de efeitos binários
netmeta (netmeta): Conduz meta-análises de rede para comparações indiretas
netmeta(TE, seTE, treat1, treat2, studlab, data = d) # estima efeitos em rede
trial_sequence (TrialEmulation): Emula ensaios clínicos a partir de dados observacionais
trial_sequence(dados) # prepara dados longitudinais para inferência causal
pmApiRequest (pubmedR): Busca e extrai metadados de artigos diretamente do PubMed
pmApiRequest(query = "RCT", limit = 100) # extrai dados bibliométricos de 100 artigos
dagitty (dagitty): Cria e analisa Grafos Acíclicos Direcionados (DAGs)
dagitty("dag { X -> Y; Z -> X; Z -> Y }") # desenha DAG para identificar confusão
survfit (survival): Calcula curvas de sobrevida (Kaplan-Meier)
survfit(Surv(tempo, evento) ~ grupo, data = d) # estima sobrevida por grupo
ggsurvplot (survminer): Plota curvas de sobrevida com tabelas de risco
ggsurvplot(fit, data = d, risk.table = TRUE) # gráfico de KM padrão científico
epi.2by2 (epiR): Calcula medidas de associação epidemiológica (RR, OR)
epi.2by2(tabela) # calcula riscos e frações atribuíveis
matchit (MatchIt): Pareamento por escore de propensão (Propensity Score)
matchit(tratam ~ idade + sexo, data = d) # reduz viés em dados observacionais
